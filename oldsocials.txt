/*
 * General social format:
 * social "command name" {
 *	  name-pt "command name in portuguese";
 *	[ hide-invis;							]
 *	[ level <level>;						]
 *	  char-pos <position>;
 *	[ vict-pos <position>;						]
 *	  char-noarg "msg sent to actor when no additional arg";
 *	[ room-noarg "msg sent to room when no additional arg";		]
 *	[ char-found "msg sent to actor when a victim is found";	]
 *	[ room-found "msg sent to room when a victim is found";		]
 *	[ vict-found "msg sent to victim when it is found";		]
 *	[ not-found "msg sent to actor when no victim is found";	]
 *	[ char-self "msg sent to actor when it actions itself";		]
 *	[ room-self "msg sent to room when actor actions itself";	]
 * };
 *
 * Command names should be written without any space, number or extended
 * character, even for portuguese names.  "abraçar" should be written
 * "abracar".  VitaliaMUD takes care of extended characters.
 * 
 * Fields surrounded by [ ] are optional. For vict-pos, it defaults to
 * DEAD.
 */
